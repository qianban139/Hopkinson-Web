## 2. 数据库 Schema

### 2.1 ER图

```
users ──┐
        │ 1:N
        ├──► experiments ──┐
        │                  │ 1:N
        │                  ├──► ai_optimizations
        │                  │ 1:1
        │                  ├──► safety_checks
        │                  │ 1:N
        │                  └──► monitor_snapshots
        │
materials (独立表，不直接关联user)
```

### 2.2 Prisma Schema

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ==================== 用户 ====================

model User {
  id          String       @id @default(uuid()) @db.Uuid
  username    String       @unique @db.VarChar(50)
  email       String       @unique @db.VarChar(255)
  password    String       @db.VarChar(255) // bcrypt hash
  role        UserRole     @default(RESEARCHER)
  createdAt   DateTime     @default(now()) @map("created_at")
  updatedAt   DateTime     @updatedAt @map("updated_at")

  experiments Experiment[]

  @@map("users")
}

enum UserRole {
  ADMIN
  RESEARCHER
  VIEWER
}

// ==================== 材料库 ====================

model Material {
  id              String   @id @default(uuid()) @db.Uuid
  name            String   @db.VarChar(100)     // 如 "Q235钢"
  category        String   @db.VarChar(50)      // metal, rock, concrete, ...
  subCategory     String?  @map("sub_category") @db.VarChar(100)
  density         Float                          // kg/m³
  elasticModulus  Float    @map("elastic_modulus") // GPa
  yieldStrength   Float    @map("yield_strength")  // MPa
  johnsonCook     Json     @map("johnson_cook")    // J-C模型参数
  stressStrainData Json?   @map("stress_strain_data") // 应力-应变数据
  metadata        Json?                           // 扩展属性
  createdAt       DateTime @default(now()) @map("created_at")
  updatedAt       DateTime @updatedAt @map("updated_at")

  experiments     Experiment[]

  @@index([category])
  @@index([name])
  @@map("materials")
}

// ==================== 实验 ====================

model Experiment {
  id              String           @id @default(uuid()) @db.Uuid
  userId          String           @map("user_id") @db.Uuid
  materialId      String           @map("material_id") @db.Uuid
  params          Json             // { voltage, current, pulseWidth, waveform }
  multiFieldParams Json?           @map("multi_field_params") // 多场耦合参数
  aiOptimized     Boolean          @default(false) @map("ai_optimized")
  results         Json?            // 实验结果
  waveformData    Json?            @map("waveform_data") // 波形数据
  status          ExperimentStatus @default(CREATED)
  createdAt       DateTime         @default(now()) @map("created_at")
  completedAt     DateTime?        @map("completed_at")

  user            User             @relation(fields: [userId], references: [id])
  material        Material         @relation(fields: [materialId], references: [id])
  optimizations   AIOptimization[]
  safetyCheck     SafetyCheck?
  snapshots       MonitorSnapshot[]

  @@index([userId])
  @@index([materialId])
  @@index([status])
  @@map("experiments")
}

enum ExperimentStatus {
  CREATED
  SAFETY_CHECK
  RUNNING
  COMPLETED
  OPTIMIZING
  COUPLING
  ANALYZING
  FAILED
}

// ==================== AI优化记录 ====================

model AIOptimization {
  id            String   @id @default(uuid()) @db.Uuid
  experimentId  String   @map("experiment_id") @db.Uuid
  algorithm     String   @db.VarChar(20) // LSTM, WGAN-GP, PPO
  inputParams   Json     @map("input_params")
  outputParams  Json     @map("output_params")
  metrics       Json     // { stressUniformity, energyEfficiency, ... }
  durationMs    Int      @map("duration_ms")
  createdAt     DateTime @default(now()) @map("created_at")

  experiment    Experiment @relation(fields: [experimentId], references: [id])

  @@index([experimentId])
  @@map("ai_optimizations")
}

// ==================== 安全检查日志 ====================

model SafetyCheck {
  id            String   @id @default(uuid()) @db.Uuid
  experimentId  String   @unique @map("experiment_id") @db.Uuid
  checklist     Json     // 7项检查结果详情
  passed        Boolean
  checkedAt     DateTime @map("checked_at")

  experiment    Experiment @relation(fields: [experimentId], references: [id])

  @@map("safety_checks")
}

// ==================== 监控快照 ====================

model MonitorSnapshot {
  id            BigInt   @id @default(autoincrement())
  experimentId  String?  @map("experiment_id") @db.Uuid
  voltage       Float    // V
  current       Float    // A
  capacitance   Float    // %
  temperature   Float    // °C
  emi           Float    // dB
  alertLevel    String   @default("normal") @map("alert_level") @db.VarChar(20)
  recordedAt    DateTime @default(now()) @map("recorded_at")

  experiment    Experiment? @relation(fields: [experimentId], references: [id])

  @@index([experimentId])
  @@index([recordedAt])
  @@map("monitor_snapshots")
}
```

## 3. API端点设计

### 3.1 RESTful API

#### 材料管理

```
GET    /api/materials
  Query: ?category=metal&search=Q235&page=1&limit=20
  Response: { data: Material[], total: number, page: number }

GET    /api/materials/:id
  Response: Material (含完整J-C参数和应力-应变数据)

POST   /api/materials                    [Admin]
  Body: { name, category, density, ... }
  Response: Material

PUT    /api/materials/:id                [Admin]
  Body: Partial<Material>
  Response: Material
```

#### 实验管理

```
POST   /api/experiments
  Body: { materialId, params }
  Response: Experiment

GET    /api/experiments/:id
  Response: Experiment (含material关联)

GET    /api/experiments
  Query: ?userId=xxx&status=COMPLETED&page=1&limit=20
  Response: { data: Experiment[], total: number }

PUT    /api/experiments/:id/params
  Body: { voltage?, current?, pulseWidth?, waveform? }
  Response: Experiment

POST   /api/experiments/:id/run
  Description: 启动实验（服务端计算波形数据）
  Response: { status: 'RUNNING', estimatedDuration: 5000 }

GET    /api/experiments/:id/waveform
  Description: 获取波形数据
  Response: { incident: [], reflected: [], transmitted: [] }
```

#### AI服务

```
POST   /api/ai/chat
  Description: AI对话（代理LLM API，隐藏API密钥）
  Body: {
    messages: [{ role, content }],
    tools?: FunctionCallingTool[],
    model?: string
  }
  Response: { message: string, toolCalls?: ToolCall[] }

POST   /api/experiments/:id/optimize
  Description: 触发AI优化
  Body: { algorithm?: 'full' | 'lstm' | 'wgan' | 'ppo' }
  Response: AIOptimization

POST   /api/ai/predict
  Description: AI材料性能预测
  Body: { materialId, strainRate, temperature }
  Response: { prediction: StressStrainCurve, confidence: number }
```

#### 安全检查

```
POST   /api/safety/check
  Description: 执行安全检查
  Body: { experimentId }
  Response: SafetyCheck

GET    /api/safety/check/:experimentId
  Response: SafetyCheck
```

#### 导出

```
POST   /api/export/csv
  Body: { experimentId, fields: string[] }
  Response: CSV文件流

POST   /api/export/pdf
  Body: { experimentId, sections: string[], template?: string }
  Response: PDF文件流
```

### 3.2 WebSocket 端点

```
WS /ws/monitor
  Description: 实时监控数据推送
  Events:
    server → client: 'monitor:update' { voltage, current, capacitance, temperature, emi }
    server → client: 'monitor:alert' { level, message, param }
    client → server: 'monitor:subscribe' { experimentId? }
  频率: 每秒1次

WS /ws/ai-progress
  Description: AI优化进度推送
  Events:
    server → client: 'ai:progress' { step, progress, metrics? }
    server → client: 'ai:complete' { result: AIOptimization }
    server → client: 'ai:error' { error: string }
    client → server: 'ai:start' { experimentId, algorithm }
    client → server: 'ai:cancel'

WS /ws/experiment
  Description: 实验执行进度
  Events:
    server → client: 'experiment:phase' { phase, progress }
    server → client: 'experiment:waveform' { timestamp, incident, reflected, transmitted }
    server → client: 'experiment:complete' { result }
```

## 5. 数据迁移方案

### 5.1 材料数据迁移

从前端 `src/data/materials.json` 迁移到 PostgreSQL：

```typescript
// server/scripts/seed-materials.ts

import { PrismaClient } from '@prisma/client';
import materialsJson from '../../src/data/materials.json';

const prisma = new PrismaClient();

async function seedMaterials() {
  for (const material of materialsJson.materials) {
    await prisma.material.upsert({
      where: { id: material.id },
      update: {},
      create: {
        id: material.id,
        name: material.name,
        category: material.category,
        subCategory: material.subCategory,
        density: material.density,
        elasticModulus: material.elasticModulus,
        yieldStrength: material.yieldStrength,
        johnsonCook: material.johnsonCook,
        stressStrainData: material.stressStrainData,
        metadata: material.metadata ?? {},
      },
    });
  }
  console.log(`Seeded ${materialsJson.materials.length} materials`);
}

seedMaterials()
  .catch(console.error)
  .finally(() => prisma.$disconnect());
```

### 5.2 迁移策略

1. **Phase 1**: 后端提供API，前端逐步切换数据来源
2. **Phase 2**: 材料库切换为API获取（保留本地JSON作为fallback）
3. **Phase 3**: 实验数据保存到数据库
4. **Phase 4**: AI对话通过后端代理（隐藏API密钥）

## 6. 部署方案

### 6.1 Docker Compose

```yaml
# docker-compose.yml

version: '3.9'

services:
  # PostgreSQL 数据库
  postgres:
    image: postgres:16-alpine
    container_name: hopkinson-db
    environment:
      POSTGRES_DB: hopkinson
      POSTGRES_USER: ${DB_USER:-hopkinson}
      POSTGRES_PASSWORD: ${DB_PASSWORD}
    volumes:
      - postgres_data:/var/lib/postgresql/data
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U hopkinson"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis 缓存
  redis:
    image: redis:7-alpine
    container_name: hopkinson-redis
    command: redis-server --appendonly yes
    volumes:
      - redis_data:/data
    ports:
      - "6379:6379"
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  # 后端 API
  api:
    build:
      context: .
      dockerfile: server/Dockerfile
    container_name: hopkinson-api
    environment:
      DATABASE_URL: postgresql://${DB_USER:-hopkinson}:${DB_PASSWORD}@postgres:5432/hopkinson
      REDIS_URL: redis://redis:6379
      DEEPSEEK_API_KEY: ${DEEPSEEK_API_KEY}
      JWT_SECRET: ${JWT_SECRET}
      NODE_ENV: production
      PORT: 3001
    ports:
      - "3001:3001"
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy

  # 前端 + Nginx
  nginx:
    image: nginx:1.25-alpine
    container_name: hopkinson-nginx
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/ssl:/etc/nginx/ssl:ro
      - ./dist:/usr/share/nginx/html:ro
    depends_on:
      - api

volumes:
  postgres_data:
  redis_data:
```

### 6.2 Nginx 配置

```nginx
# nginx/nginx.conf

upstream api_backend {
    server api:3001;
}

server {
    listen 80;
    server_name hopkinson.example.com;

    # 重定向到HTTPS
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    server_name hopkinson.example.com;

    ssl_certificate     /etc/nginx/ssl/cert.pem;
    ssl_certificate_key /etc/nginx/ssl/key.pem;

    # 前端静态资源
    root /usr/share/nginx/html;
    index index.html;

    # SPA路由回退
    location / {
        try_files $uri $uri/ /index.html;
    }

    # API反向代理
    location /api/ {
        proxy_pass http://api_backend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # WebSocket反向代理
    location /ws/ {
        proxy_pass http://api_backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_read_timeout 86400;
    }

    # 视频静态资源（启用范围请求+缓存）
    location /videos/ {
        expires 30d;
        add_header Cache-Control "public, immutable";
        add_header Accept-Ranges bytes;
    }

    # 静态资源缓存
    location ~* \.(js|css|png|jpg|svg|woff2)$ {
        expires 7d;
        add_header Cache-Control "public, immutable";
    }

    # Gzip压缩
    gzip on;
    gzip_types text/plain application/json application/javascript text/css image/svg+xml;
    gzip_min_length 1024;
}
```

## 9. 安全措施

| 措施 | 实现方式 |
|------|----------|
| API密钥保护 | LLM调用通过后端代理，密钥不暴露给前端 |
| 认证 | JWT Token，httpOnly cookie |
| 密码存储 | bcrypt (cost=12) |
| 输入验证 | zod schema验证所有请求体 |
| SQL注入 | Prisma ORM参数化查询 |
| 速率限制 | express-rate-limit（100请求/分钟/IP） |
| CORS | 仅允许指定域名 |
| HTTPS | Nginx SSL终止 |
| 日志 | 敏感信息脱敏 |
| 环境变量 | dotenv + Docker secrets |
